        return chunks
    
    def clear_cache(self):
        self.cache.clear()

    """–í–∞–ª–∏–¥–∞—Ü–∏—è —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤ —Ä–∞–∑–¥–µ–ª–µ–Ω–∏—è"""

    for i, chunk in enumerate(chunks):
        # –ü—Ä–æ–≤–µ—Ä–∫–∞ —Ä–∞–∑–º–µ—Ä–∞
        if len(chunk) < min_size:
            issues.append(f"–ß–∞–Ω–∫ {i} —Å–ª–∏—à–∫–æ–º –º–∞–ª: {len(chunk)} —Å–∏–º–≤–æ–ª–æ–≤")
        
        if len(chunk) > max_size:

        # –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –ø—É—Å—Ç—ã–µ —á–∞–Ω–∫–∏
        if not chunk.strip():
            issues.append(f"–ß–∞–Ω–∫ {i} –ø—É—Å—Ç–æ–π –∏–ª–∏ —Å–æ–¥–µ—Ä–∂–∏—Ç —Ç–æ–ª—å–∫–æ –ø—Ä–æ–±–µ–ª—ã")
        
        # –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –æ–±—Ä—ã–≤—ã –ø—Ä–µ–¥–ª–æ–∂–µ–Ω–∏–π
        if not chunk.strip().endswith(('.', '!', '?', '\n')):
            issues.append(f"–ß–∞–Ω–∫ {i} –º–æ–∂–µ—Ç —Å–æ–¥–µ—Ä–∂–∞—Ç—å –æ–±—Ä—ã–≤—ã –ø—Ä–µ–¥–ª–æ–∂–µ–Ω–∏–π")
    
    return issues
